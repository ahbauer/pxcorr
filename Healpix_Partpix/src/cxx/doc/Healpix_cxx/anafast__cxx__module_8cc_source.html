00052 template<typename T> void anafast_cxx (paramfile &params)
00053   {
00054   int nlmax = params.template find<int>("nlmax");
00055   int nmmax = params.template find<int>("nmmax",nlmax);
00056   string infile = params.template find<string>("infile");
00057   string outfile = params.template find<string>("outfile","");
00058   string outfile_alms = params.template find<string>("outfile_alms","");
00059   planck_assert ((outfile!="") || (outfile_alms!=""),
00060     "no output specified, nothing done");
00061   bool polarisation = params.template find<bool>("polarisation");
00062   int num_iter = params.template find<int>("iter_order",0);
00063 
00064   if (!polarisation)
00065     {
00066     Healpix_Map<T> map;
00067     read_Healpix_map_from_fits(infile,map,1,2);
00068     arr<double> weight;
00069     get_ring_weights (params,map.Nside(),weight);
00070 
00071     Alm<xcomplex<T> > alm(nlmax,nmmax);
00072     double avg=map.average();
00073     map.Add(T(-avg));
00074     if (map.Scheme()==NEST) map.swap_scheme();
00075     map2alm_iter(map,alm,num_iter,weight);
00076 
00077     alm(0,0) += T(avg*sqrt(fourpi));
00078 
00079     if (outfile!="")
00080       {
00081       PowSpec powspec;
00082       extract_powspec (alm,powspec);
00083       write_powspec_to_fits (outfile,powspec,1);
00084       }
00085     if (outfile_alms!="")
00086       write_Alm_to_fits(outfile_alms,alm,nlmax,nmmax,planckType<T>());
00087     }
00088   else
00089     {
00090     Healpix_Map<T> mapT, mapQ, mapU;
00091     read_Healpix_map_from_fits(infile,mapT,mapQ,mapU,2);
00092     arr<double> weight;
00093     get_ring_weights (params,mapT.Nside(),weight);
00094 
00095     Alm<xcomplex<T> > almT(nlmax,nmmax), almG(nlmax,nmmax), almC(nlmax,nmmax);
00096     double avg=mapT.average();
00097     mapT.Add(T(-avg));
00098     if (mapT.Scheme()==NEST) mapT.swap_scheme();
00099     if (mapQ.Scheme()==NEST) mapQ.swap_scheme();
00100     if (mapU.Scheme()==NEST) mapU.swap_scheme();
00101     map2alm_pol_iter
00102       (mapT,mapQ,mapU,almT,almG,almC,num_iter,weight);
00103 
00104     almT(0,0) += T(avg*sqrt(fourpi));
00105 
00106     if (outfile!="")
00107       {
00108       PowSpec powspec;
00109       extract_powspec (almT,almG,almC,powspec);
00110       params.template find<bool>("full_powerspectrum",false) ?
00111         write_powspec_to_fits (outfile,powspec,6) :
00112         write_powspec_to_fits (outfile,powspec,4);
00113       }
00114     if (outfile_alms!="")
00115       write_Alm_to_fits (outfile_alms,almT,almG,almC,nlmax,nmmax,planckType<T>());
00116     }
00117   }
