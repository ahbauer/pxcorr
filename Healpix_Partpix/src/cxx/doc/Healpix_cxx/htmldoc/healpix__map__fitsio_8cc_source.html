00040 unsigned int healpix_repcount (tsize npix)
00041   {
00042   if (npix<1024) return 1;
00043   if ((npix%1024)==0) return 1024;
00044   return isqrt (npix/12);
00045   }

00049 template<typename T> void read_Healpix_map_from_fits
00050   (fitshandle &inp, Healpix_Map<T> &map, int colnum)
00051   {
00052   arr<T> myarr;
00053   inp.read_entire_column (colnum, myarr);
00054   int64 nside = inp.get_key<int>("NSIDE");
00055   planck_assert (int64(myarr.size())==12*nside*nside,
00056     string("mismatch between number of map pixels ("
00057     +dataToString(myarr.size())+") and Nside ("+dataToString(nside)+")"));
00058   map.Set (myarr, string2HealpixScheme(inp.get_key<string>("ORDERING")));
00059   }

00086 void prepare_Healpix_fitsmap
00087   (fitshandle &out, const Healpix_Base &base, PDT datatype,
00088   const arr<string> &colname)
00089   {
00090   vector<fitscolumn> cols;
00091   int repcount = healpix_repcount (base.Npix());
00092   for (tsize m=0; m<colname.size(); ++m)
00093     cols.push_back (fitscolumn (colname[m],"unknown",repcount, datatype));
00094   out.insert_bintab(cols);
00095   out.set_key ("PIXTYPE",string("HEALPIX"),"HEALPIX pixelisation");
00096   string ordering = (base.Scheme()==RING) ? "RING" : "NESTED";
00097   out.set_key ("ORDERING",ordering,
00098                   "Pixel ordering scheme, either RING or NESTED");
00099   out.set_key ("NSIDE",base.Nside(),"Resolution parameter for HEALPIX");
00100   out.set_key ("FIRSTPIX",0,"First pixel # (0 based)");
00101   out.set_key ("LASTPIX",base.Npix()-1,"Last pixel # (0 based)");
00102   out.set_key ("INDXSCHM",string("IMPLICIT"),
00103                   "Indexing: IMPLICIT or EXPLICIT");
00104   }
