00043 template<typename T> void udgrade_cxx (paramfile &params)
00044   {
00045   string infile = params.template find<string>("infile");
00046   string outfile = params.template find<string>("outfile");
00047   int order = Healpix_Base::nside2order (params.template find<int>("nside"));
00048   bool polarisation = params.template find<bool>("polarisation",false);
00049   bool pessimistic = params.template find<bool>("pessimistic",false);
00050 
00051   if (!polarisation)
00052     {
00053     Healpix_Map<T> inmap;
00054     read_Healpix_map_from_fits(infile,inmap,1,2);
00055     Healpix_Map<T> outmap (order, inmap.Scheme());
00056 
00057     outmap.Import(inmap,pessimistic);
00058     write_Healpix_map_to_fits (outfile,outmap,planckType<T>());
00059     }
00060   else
00061     {
00062     Healpix_Map<T> inmap;
00063     read_Healpix_map_from_fits(infile,inmap,1,2);
00064     Healpix_Map<T> outmapT (order, inmap.Scheme()),
00065                    outmapQ (order, inmap.Scheme()),
00066                    outmapU (order, inmap.Scheme());
00067 
00068     outmapT.Import(inmap,pessimistic);
00069     read_Healpix_map_from_fits(infile,inmap,2,2);
00070     outmapQ.Import(inmap,pessimistic);
00071     read_Healpix_map_from_fits(infile,inmap,3,2);
00072     outmapU.Import(inmap,pessimistic);
00073     write_Healpix_map_to_fits (outfile,outmapT,outmapQ,outmapU,planckType<T>());
00074     }
00075   }

00077 } // unnamed namespace
