00041 class paramfile
00042   {
00043   private:

00045     params_type params;
00046     mutable std::set<std::string> read_params;

00049     std::string get_valstr(const std::string &key) const
00050       {
00051       params_type::const_iterator loc=params.find(key);
00052       if (loc!=params.end()) return loc->second;
00053       planck_fail ("Cannot find the key '" + key + "'.");
00054       }

00056     bool param_unread (const std::string &key) const
00057       { return (read_params.find(key)==read_params.end()); }

00059   public:

00061       : verbose(verbose_)
00062       { parse_file (filename, params); }

00064     paramfile (const params_type &par, bool verbose_=true)
00065       : params (par), verbose(verbose_)
00066       {}
00067 
00068     ~paramfile ()
00069       {
00070       if (verbose)
00071         for (params_type::const_iterator loc=params.begin();
00072              loc!=params.end(); ++loc)
00073           if (param_unread(loc->first))
00074             std::cout << "Parser warning: unused parameter '"
00075                       << loc->first << "'" << std::endl;
00076       }
00077 
00078     void setVerbosity (bool verbose_)
00079       { verbose = verbose_; }
00080 
00081     bool getVerbosity () const
00082       { return verbose; }

00084     bool param_present(const std::string &key) const
00085       { return (params.find(key)!=params.end()); }

00087     template<typename T> T find (const std::string &key) const
00088       {
00089       T result;
00090       stringToData(get_valstr(key),result);
00091       std::string output = dataToString(result);
00092       if (planckType<T>()==PLANCK_STRING) output = "'"+output+"'";
00093       if (verbose && param_unread(key))
00094         std::cout << "Parser: " << key << " = " << output << std::endl;
00095       read_params.insert(key);
00096       return result;
00097       }
00098     template<typename T> T find
00099       (const std::string &key, const T &deflt)
00100       {
00101       if (param_present(key)) return find<T>(key);
00102       std::string output = dataToString(deflt);
00103       if (planckType<T>()==PLANCK_STRING) output = "'"+output+"'";
00104       if (verbose && param_unread(key))
00105         std::cout << "Parser: " << key << " = " << output
00106                   << " <default>" << std::endl;
00107       params[key]=dataToString(deflt);
00108       read_params.insert(key);
00109       return deflt;
00110       }
