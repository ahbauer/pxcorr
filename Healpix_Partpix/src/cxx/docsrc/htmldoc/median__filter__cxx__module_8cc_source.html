00043 template<typename Iterator> typename iterator_traits<Iterator>::value_type
00044   median(Iterator first, Iterator last)
00045   {
00046   Iterator mid = first+(last-first-1)/2;
00047   nth_element(first,mid,last);
00048   if ((last-first)&1) return *mid;
00049   return typename iterator_traits<Iterator>::value_type
00050     (0.5*((*mid)+(*min_element(mid+1,last))));
00051   }
00052 
00053 } // unnamed namespace

00055 int median_filter_cxx_module (int argc, const char **argv)
00056   {
00057   module_startup ("median_filter_cxx", argc, argv, 4,
00058                   "<input map> <output map> <radius in arcmin>");
00059 
00060   double radius = stringToData<double>(argv[3])*arcmin2rad;
00061 
00062   Healpix_Map<float> inmap;
00063   read_Healpix_map_from_fits(argv[1],inmap,1,2);
00064   Healpix_Map<float> outmap (inmap.Nside(), inmap.Scheme(), SET_NSIDE);
00065 
00066   vector<int> list;
00067   vector<float> list2;
00068   for (int m=0; m<inmap.Npix(); ++m)
00069     {
00070     inmap.query_disc(inmap.pix2ang(m),radius,list);
00071     list2.resize(list.size());
00072     for (tsize i=0; i<list.size(); ++i)
00073       list2[i] = inmap[list[i]];
00074     outmap[m]=median(list2.begin(),list2.end());
00075     }
00076 
00077   write_Healpix_map_to_fits (argv[2],outmap,PLANCK_FLOAT32);
00078 
00079   return 0;
